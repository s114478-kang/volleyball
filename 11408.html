let bomb;
let min;
let max;
let currentPlayer;
let gameOver;

function startGame() {
    bomb = Math.floor(Math.random() * 99) + 2; // 2~100
    min = 1;
    max = 100;
    currentPlayer = 1;
    gameOver = false;

    document.getElementById("min").textContent = min;
    document.getElementById("max").textContent = max;
    document.getElementById("history").innerHTML = "";
    document.getElementById("message").innerHTML = "";
    document.getElementById("playerText").textContent =
        `玩家 ${currentPlayer} 的回合`;
    document.getElementById("guessInput").value = "";

    console.log("炸彈數字:", bomb);
}

function guess() {
    if (gameOver) return;

    const input = document.getElementById("guessInput");
    const num = Number(input.value);

    if (!Number.isInteger(num)) {
        showMessage("請輸入整數", "orange");
        return;
    }

    if (num <= min || num >= max) {
        showMessage(
            `請輸入 ${min} 與 ${max} 之間的數字`,
            "orange"
        );
        return;
    }

    addHistory(`玩家 ${currentPlayer} 猜了 ${num}`);

    if (num === bomb) {
        gameOver = true;

        document.getElementById("message").innerHTML =
            `<span class="win">💥 玩家 ${currentPlayer} 踩到炸彈！</span>`;

        addHistory(
            `💥 玩家 ${currentPlayer} 猜中炸彈 (${bomb})`
        );

        return;
    }

    if (num < bomb) {
        min = num;
        showMessage("太小了！", "#2196F3");
    } else {
        max = num;
        showMessage("太大了！", "#2196F3");
    }

    document.getElementById("min").textContent = min;
    document.getElementById("max").textContent = max;

    currentPlayer = currentPlayer === 1 ? 2 : 1;

    document.getElementById("playerText").textContent =
        `玩家 ${currentPlayer} 的回合`;

    input.value = "";
    input.focus();
}

function showMessage(text, color) {
    const msg = document.getElementById("message");
    msg.textContent = text;
    msg.style.color = color;
}

function addHistory(text) {
    const history = document.getElementById("history");

    const div = document.createElement("div");
    div.className = "record";
    div.textContent = text;

    history.prepend(div);
}

document

    .addEventListener("keydown", function(e){
        if(e.key === "Enter"){
            guess();
        }
    });

startGame();
</script>

<script>
const player = document.getElementById("player");
const ball = document.getElementById("ball");
const scoreText = document.getElementById("score");
const message = document.getElementById("message");

const groundY = window.innerHeight - 120;

let score = 0;

/* 玩家 */
let px = 180;
let py = groundY - 100;
let vy = 0;

let jumping = false;
let spiked = false;

/* 球 */
let bx = 320;
let by = 260;

let bvx = -2;
let bvy = 0;

const gravity = 0.45;

/* 重置球 */
function resetBall(){
    bx = 320;
    by = 260;
    bvx = -2;
    bvy = 0;
    spiked = false;
}

/* 顯示訊息 */
function showMessage(text){
    message.textContent = text;

    setTimeout(()=>{
        if(message.textContent===text){
            message.textContent="";
        }
    },1200);
}

/* 空白鍵 */
document.addEventListener("keydown",(e)=>{

    if(e.code !== "Space") return;

    e.preventDefault();

    if(!jumping){
        jumping = true;
        vy = -14;
    }
    else{
        trySpike();
    }
});

/* 扣球 */
function trySpike(){

    if(spiked) return;

    const playerCenterX = px + 30;
    const playerCenterY = py + 20;

    const dx = bx - playerCenterX;
    const dy = by - playerCenterY;

    const dist = Math.sqrt(dx*dx + dy*dy);

    if(dist < 100){

        bvx = 10;
        bvy = -8;

        spiked = true;

        showMessage("扣球！");
    }
}

/* 更新 */
function update(){

    /* 玩家 */
    vy += gravity;
    py += vy;

    if(py >= groundY - 100){
        py = groundY - 100;
        vy = 0;
        jumping = false;
    }

    /* 球 */
    bvy += gravity;
    bx += bvx;
    by += bvy;

    /* 左牆 */
    if(bx < 0){
        bx = 0;
        bvx *= -1;
    }

    /* 右牆 */
    if(bx > window.innerWidth - 36){
        bx = window.innerWidth - 36;
        bvx *= -1;
    }

    /* 球網 */
    const netX = window.innerWidth / 2;

    if(
        bx + 36 > netX - 5 &&
        bx < netX + 5 &&
        by > groundY - 340
    ){
        if(bvx > 0){
            bx = netX - 36;
        }else{
            bx = netX + 10;
        }

        bvx *= -0.8;
    }

    /* 地面 */
    if(by > groundY - 36){

        if(bx > netX){
            score++;
            scoreText.textContent = "得分：" + score;
            showMessage("得分！");
        }
        else{
            showMessage("失敗！");
        }

        resetBall();
    }

    /* 玩家位置 */
    player.style.left = px + "px";
    player.style.top = py + "px";

    /* 球位置 */
    ball.style.left = bx + "px";
    ball.style.top = by + "px";

    requestAnimationFrame(update);
}

update();
</script>

const ball = document.getElementById("ball");
const scoreText = document.getElementById("score");
const bestText = document.getElementById("best");
const message = document.getElementById("message");
const restartBtn = document.getElementById("restart");

const groundY = window.innerHeight - 120;

const playerX = window.innerWidth / 2;
const playerTop = groundY - 110;

let ballX;
let ballY;
let vx;
let vy;

let score = 0;
let best = 0;
let gameOver = false;

function resetGame(){

    score = 0;
    gameOver = false;

    scoreText.textContent = "分數：0";
    message.textContent = "";

    restartBtn.style.display = "none";

    ballX = playerX - 20;
    ballY = 220;

    vx = (Math.random() * 4) - 2;
    vy = 0;
}

function setBall(){

    ball.style.left = ballX + "px";
    ball.style.top = ballY + "px";
}

function gameEnd(){

    gameOver = true;

    if(score > best){
        best = score;
        bestText.textContent = "最高紀錄：" + best;
    }

    message.textContent = "球落地了！";
    restartBtn.style.display = "block";
}

document.addEventListener("keydown", e=>{

    if(e.code !== "Space") return;

    e.preventDefault();

    if(gameOver) return;

    const ballCenterX = ballX + 20;
    const ballCenterY = ballY + 20;

    const targetX = playerX;
    const targetY = playerTop + 20;

    const dx = ballCenterX - targetX;
    const dy = ballCenterY - targetY;

    const distance = Math.sqrt(dx*dx + dy*dy);

    if(distance < 80){

        score++;

        scoreText.textContent = "分數：" + score;

        vy = -12 - Math.min(score * 0.1, 5);

        vx += (Math.random()-0.5)*2;

        message.textContent = "漂亮托球！";

        clearTimeout(window.msgTimer);

        window.msgTimer = setTimeout(()=>{
            if(!gameOver){
                message.textContent = "";
            }
        },500);
    }
});

restartBtn.addEventListener("click", resetGame);

function update(){

    if(!gameOver){

        vy += 0.35;

        ballX += vx;
        ballY += vy;

        if(ballX < 0){
            ballX = 0;
            vx *= -1;
        }

        if(ballX > window.innerWidth - 40){
            ballX = window.innerWidth - 40;
            vx *= -1;
        }

        if(ballY >= groundY - 40){
            gameEnd();
        }
    }

}

resetGame();
